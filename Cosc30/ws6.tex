\begin{problem}{1}
    Provide a good upper bound for the following recurrence. For all problems below
    assume $T(0) = T(1) = 1$.
    \begin{enumerate}
        \item $T(n) \le 4 \cdot T(n-2) + 1 \quad \text{if } n > 1.$
        \item $T(n) \le T\!\left(\left\lfloor \frac{n}{3} \right\rfloor\right) + n \quad \text{if } n > 1.$
        \item Let $C$ be a positive constant. \[T(n) \le 3 \cdot T\!\left(\left\lfloor \frac{2n}{3} \right\rfloor\right) + C \quad \text{if } n > 1. \]
    \end{enumerate}
\end{problem}

\begin{solution}
    \bbni
    \begin{enumerate}
        \item $T(n) \le 4 \cdot T(n-2) + 1$ 
        
        We prove that $T(n) \leq \dfrac{4^{\floor{\frac{n}{2}} + 1 } -1}{ 3}$ for all $n \geq 1$ using induction. Let $n$ be an arbitrary positive integer. Assume $T(k) \leq \dfrac{4^{\floor{\frac{k}{2}} + 1 } -1}{ 3}$ for all $k$ less than $n$. We check the following cases: 
        \begin{enumerate}
            \item If $n \geq 1$, we can upper bound $T(n)$ by $4 \cdot T(n-2) + 1$. Since $n-2 < n$, we substitute, 
            \begin{align*}
                T(n) &\leq 4 \cdot T(n-2)  + 1  \\ 
                &= 4 \cdot \dfrac{4^{\floor{\frac{n-2}{2}} + 1 } -1}{ 3} + 1 \\
                &= \dfrac{4^{\floor{\frac{n-2}{2}} + 2 } -4 + 3}{ 3} \\ 
                &= \dfrac{4^{\floor{\frac{n}{2}} + 1 } - 1 }{ 3} 
            \end{align*} 
            Hence, $T(n) \leq  \dfrac{4^{\floor{\frac{n-2}{2}} + 1 } -1}{ 3}$ holds for $n > 1$. 
            \item If $n = 1$, we get 
            \[T(1) =  \dfrac{4^{\floor{\frac{1}{2}} + 1 } -1}{ 3}  = \dfrac{4^{1} -1}{3} = \dfrac{3}{3} = 1 \]
            So the base case holds. 
        \end{enumerate}
        Therefore the statement is proved. 
        
        Note: We use the floor function $ k = \floor{\frac{n}{2}}$ so that $n - 2k \in \{0, 1\}$, and odd for $n$ we get there one step quicker.
        
        \item $T(n) \le T\!\left(\left\lfloor \frac{n}{3} \right\rfloor\right) + n$
        
        We prove that $T(n) \leq \dfrac{3n-1}{2}$ by induction for all $n \geq 1$. Let $n$ be an arbitrary positive integer. Assume $T(k) \leq \dfrac{3n-1}{2}$ for all $k$ less than $n$. We check the following cases: 
        \begin{enumerate}
            \item If $n \geq 1$, we can upper bound $T(n)$ by $T(\floor{\frac{n}{3}}) + n$. Since $\floor{\frac{n}{3}} < n$, we substitute, 
            \begin{align*}
                T(n) &\leq T\left (\floor{\frac{n}{3}} \right ) + n \\ 
                &= \dfrac{3(\floor{\frac{n}{3}})-1}{2} + n \\ 
                &= \dfrac{n-1}{2} + n \\ 
                &= \frac{3n - 1}{2}
            \end{align*} 
            Hence, $T(n)  \leq \dfrac{3n-1}{2}$ holds for $n > 1$. 

            \item If $n = 1$, we get 
            \[T(1) =  \dfrac{3-1}{2}  = 1\]
            So the base case holds. 
        \end{enumerate}
        Therefore the statement is proved. 

    \item $T(n) \le 3 \cdot T\!\left(\left\lfloor \frac{2n}{3} \right\rfloor\right) + C \quad \text{if } n > 1,$ for some positive constant $C$
    
    We prove that $T(n) \leq 3^{(log_{3/2}n)} + C \cdot \dfrac{3^{(log_{3/2}n)} - 1}{2}$ by induction for all $n \geq 1$.  Let $n$ be an arbitrary positive integer. Assume $T(k) \leq 3^{(log_{3/2}k)} + C \cdot \dfrac{3^{(log_{3/2}k)} - 1}{2}$ for all $k$ less than $n$. We check the following cases: 
        \begin{enumerate}
            \item If $n \geq 1$, we can upper bound $T(n)$ by $3 \cdot T(\floor{\frac{2n}{3}}) + C$. Since $\floor{\frac{2n}{3}} < n$, we substitute, 
            \begin{align*}
                T(n) &\leq 3 \cdot T \left (\floor{\frac{2n}{3}} \right ) + C \\ 
                &=  3 \cdot \left(3^{
                    \left (log_{3/2}{\floor{\frac{2n}{3}}} \right )} + C \cdot \dfrac{3^{\left (log_{3/2}\floor{{\frac{2n}{3}}} \right)} - 1}{2}\right ) + C\\ 
            \end{align*} 
            But, 
            \begin{align*}
                log_{3/2}\floor{\frac{2n}{3}} &\leq log_{3/2}\fr{2n}{3}\\ 
            \intertext{Using the upper bound,}
                T(n) &\leq 3 \cdot T\left (\floor{\frac{2n}{3}} \right ) + C \\ 
                &=  3 \cdot \left(3^{
                    \left (log_{3/2}{{\frac{2n}{3}}} \right )} + C \cdot \dfrac{3^{\left (log_{3/2}{{\frac{2n}{3}}} \right)} - 1}{2}\right ) + C\\ 
                &=  3 \cdot \left(3^{
                    \left (log_{3/2}{{n}} \right ) -1} + C \cdot \dfrac{3^{\left(log_{3/2}n \right) -1 } - 1}{2} \right) + C\\ 
                &=  \left(3^{
                    \left (log_{3/2}{{n}} \right )} + C \cdot \dfrac{3^{\left(log_{3/2}n \right) } - 3}{2} \right) + C\\ 
                 &=  3^{
                    \left (log_{3/2}{{n}} \right )} + C \cdot \left( \dfrac{3^{\left(log_{3/2}n \right) } - 1}{2} \right)\\ 
            \end{align*}
            Hence, $T(n)  \leq  3^{
                    \left (log_{3/2}{{n}} \right )} + C \cdot \left( \dfrac{3^{\left(log_{3/2}n \right) } - 1}{2} \right)$ holds for $n > 1$. 

            \item If $n = 1$, we get 
            \[T(1) =   3^{0} + C \cdot \left( \dfrac{3^0 - 1}{2} \right) = 1\]
            So the base case holds. 
        \end{enumerate}
        Therefore the statement is proved. 

    \end{enumerate}
\end{solution}
